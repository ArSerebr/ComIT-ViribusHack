# HTTP -> HTTPS redirect
server {
    listen 80;
    server_name 89-179-244-0.static.corbina.ru;

    # Для Let's Encrypt
    location /.well-known/acme-challenge/ {
        root /var/www/certbot;
    }

    location / {
        return 301 https://$server_name$request_uri;
    }
}

server {
    listen 7007 ssl;
    # http2 on;
    server_name 89-179-244-0.static.corbina.ru;

    ssl_certificate /etc/letsencrypt/live/89-179-244-0.static.corbina.ru/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/89-179-244-0.static.corbina.ru/privkey.pem;

    location / {
        proxy_pass http://127.0.0.1:7008;

        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
        proxy_buffering off;
        proxy_read_timeout 86400;
    }
}
